How to test - Mocking
        
        use Test2::Tools::Mock;
        
        my $mock = mock 'Some::Class' => (
            add => [
                new_method => sub { ... },
            ],
            override => [
                replace_method => sub { ... },
            ],
        );
         
        Some::Class->new_method();        # Calls the newly injected method
        Some::Class->replace_method();    # Calls our replacement method.
